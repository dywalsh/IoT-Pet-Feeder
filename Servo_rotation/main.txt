/**
 * @file Sample app to demonstrate PWM.
 *
 * This app uses PWM[0].
 */

#include <zephyr.h>
//#include <misc/printk.h>
#include <device.h>
#include <pwm.h>
#include <gpio.h>
#include <nrf_gpio.h>
#include <inttypes.h>

#include <board.h>
#define PWM_DRIVER CONFIG_PWM_NRF5_SW_0_DEV_NAME
#define PWM_CHANNEL LED0_GPIO_PIN
#define YOUR_PIN	NRF_GPIO_PIN_MAP(0,22)
#define PORT	CONFIG_GPIO_NRF5_P0_DEV_NAME
#define PIN		11

#define BTN_PORT	SW0_GPIO_NAME
#define BTN			SW0_GPIO_PIN

struct device *pwm_dev;
struct device *btn_dev;
struct gpio_callback btn_callback;

/*
 * 50 is flicker fusion threshold. Modulated light will be perceived
 * as steady by our eyes when blinking rate is at least 50.
 */
#define PERIOD (USEC_PER_SEC / 50)

void delay(unsigned int mseconds)
{
	u32_t goal = mseconds*1000;
	u32_t t = k_cycle_get_32();
    while ( goal > SYS_CLOCK_HW_CYCLES_TO_NS( k_cycle_get_32() - t ) );
	printf("%" PRIu32 "\n", SYS_CLOCK_HW_CYCLES_TO_NS( k_cycle_get_32() - t ));
}

void turn_clockwise (){
		u32_t pulse_width = PERIOD;
		int i = 0;
		while(i<3){
			if (pwm_pin_set_usec(pwm_dev, YOUR_PIN, PERIOD, pulse_width)) {
				printk("pwm pin set fails1\n");
				return;
			}
			pulse_width -=200;
			delay(180);
			i++;
		}
}

void turn_anticlockwise (){
		u32_t pulse_width = 0;
		int i = 0;
		while(i<3){
			if (pwm_pin_set_usec(pwm_dev, YOUR_PIN, PERIOD, pulse_width)) {
				printk("pwm pin set fails2\n");
				return;
			}
			pulse_width +=200;
			delay(180);
			i++;
		}
}

void fill_up(){
	turn_clockwise(pwm_dev);
	pwm_pin_set_usec(pwm_dev, YOUR_PIN, 0, 0);
	delay(MSEC_PER_SEC);
	turn_anticlockwise(pwm_dev);
	pwm_pin_set_usec(pwm_dev, YOUR_PIN, 0, 0);
}

void btn_handler(struct device *port, struct gpio_callback *cb,
					u32_t pins)
{
	printk("Button pressed\n");
	fill_up();
}

void main(void)
{
	//setup
	//struct device *pwm_dev;
	//struct device *btn_dev;
	u32_t cur_val;
	u32_t last_val = 1;
	btn_dev = device_get_binding(BTN_PORT);
	gpio_pin_configure(btn_dev, BTN, GPIO_DIR_IN | GPIO_INT
		| GPIO_INT_ACTIVE_LOW | GPIO_INT_EDGE | GPIO_PUD_PULL_UP);
	printk("PWM servo control 90degree clockwise -> 90degree anticlockwise\n");
	pwm_dev = device_get_binding(PWM_DRIVER);
	if (!pwm_dev) {
		printk("Cannot find %s!\n", PWM_DRIVER);
		return;
	}

	while (1) {
		//Turn motor on button press
		gpio_pin_read(btn_dev, BTN, &cur_val);
		if (cur_val == 0 && last_val == 1) {
			printk("Button press detected ");
			fill_up(pwm_dev);
		}
		last_val = cur_val;		
	}
	
	/*gpio_init_callback(&btn_callback, btn_handler, BIT(BTN));
	gpio_add_callback(btn_dev, &btn_callback);
	gpio_pin_enable_callback(btn_dev, BTN);*/
}
